## 🕹️ Mini Game — Catch the Commits!

<details>
<summary>🎮 <b>Click to reveal — drop the HTML in a GitHub Pages site and link it here!</b></summary>

```html
<!DOCTYPE html>
<html>
<head>
<title>Catch the Commits — Arjun's Arcade</title>
<style>
  body{margin:0;background:#0d1117;display:flex;flex-direction:column;align-items:center;
       justify-content:center;height:100vh;font-family:'Fira Code',monospace;color:#e6edf3}
  canvas{border:1px solid #21262d;border-radius:8px;background:#161b22}
  #info{margin-top:12px;font-size:13px;color:#8b949e;text-align:center}
  #score{font-size:16px;color:#58a6ff;margin-bottom:8px}
  h2{color:#58a6ff;font-size:18px;margin-bottom:4px}
</style>
</head>
<body>
<h2>⚡ Catch the Commits</h2>
<div id="score">score: 0 | lives: ❤❤❤</div>
<canvas id="c" width="600" height="300"></canvas>
<div id="info">click canvas to start · ← → arrow keys to move<br>
catch ● commits (+10) · ⚡ bonus (+20) · avoid 🐛 bugs (-15)</div>
<script>
var c=document.getElementById('c'),ctx=c.getContext('2d');
var p={x:270,y:270,w:60,h:10,spd:0},items=[],score=0,lives=3,run=false,frame=0;
var types=[{l:'●',pt:10},{l:'⚡',pt:20},{l:'🐛',pt:-15}];
document.addEventListener('keydown',function(e){
  if(e.key==='ArrowLeft')p.spd=-5;
  if(e.key==='ArrowRight')p.spd=5;
  if(!run){run=true;loop()}
});
document.addEventListener('keyup',function(e){
  if(e.key==='ArrowLeft'||e.key==='ArrowRight')p.spd=0;
});
c.addEventListener('click',function(){if(!run){run=true;loop()}});
c.addEventListener('touchstart',function(e){
  if(!run){run=true;loop();return}
  p.spd=e.touches[0].clientX<c.getBoundingClientRect().left+c.width/2?-5:5;
},{passive:true});
c.addEventListener('touchend',function(){p.spd=0},{passive:true});
function loop(){
  if(!run)return;
  frame++;
  ctx.clearRect(0,0,600,300);
  p.x=Math.max(0,Math.min(540,p.x+p.spd));
  ctx.fillStyle='#58a6ff';
  ctx.beginPath();ctx.roundRect(p.x,p.y,p.w,p.h,4);ctx.fill();
  if(frame%50===0){
    var ti=Math.random()<0.25?2:Math.random()<0.5?1:0;
    items.push({x:20+Math.random()*560,y:-20,vy:2.5+score/200,t:types[ti]});
  }
  items=items.filter(function(it){
    it.y+=it.vy;
    ctx.font='18px monospace';ctx.textAlign='center';
    ctx.fillText(it.t.l,it.x,it.y);
    if(it.y>300)return false;
    if(it.x>p.x&&it.x<p.x+p.w&&it.y>p.y&&it.y<p.y+p.h){
      score=Math.max(0,score+it.t.pt);
      if(it.t.pt<0){
        lives--;
        if(lives<=0){
          run=false;
          ctx.fillStyle='#e6edf3';ctx.font='22px monospace';ctx.textAlign='center';
          ctx.fillText('game over · score: '+score,300,150);
          ctx.font='14px monospace';ctx.fillStyle='#8b949e';
          ctx.fillText('refresh to play again',300,180);
        }
      }
      document.getElementById('score').textContent='score: '+score+' | lives: '+'❤'.repeat(lives);
      return false;
    }
    return true;
  });
  requestAnimationFrame(loop);
}
ctx.fillStyle='#8b949e';ctx.font='14px monospace';ctx.textAlign='center';
ctx.fillText('click to start',300,150);
</script>
</body>
</html>
```

**Deploy steps:**
1. Create repo: `ArjunKalirana/ArjunKalirana.github.io`
2. Paste the HTML above as `index.html`
3. Go to repo Settings → Pages → Source: `main` branch
4. Your game is live at `https://arjunkalirana.github.io` 🎮

<summary>⚙️ <b>Setup guide for dynamic features (read me first!)</b></summary>

### 1. WakaTime auto-metrics
- Sign up at [wakatime.com](https://wakatime.com) and install their VS Code plugin
- Add secrets to your repo: `WAKATIME_API_KEY` and `GH_TOKEN`
- Create `.github/workflows/waka.yml`:

```yaml
name: Waka Readme
on:
  schedule: [{ cron: "0 0 * * *" }]
  workflow_dispatch:
jobs:
  update-readme:
    runs-on: ubuntu-latest
    steps:
      - uses: anmol098/waka-readme-stats@master
        with:
          WAKATIME_API_KEY: ${{ secrets.WAKATIME_API_KEY }}
          GH_TOKEN: ${{ secrets.GH_TOKEN }}
          SHOW_OS: "False"
          SHOW_PROJECTS: "False"
          SHOW_COMMIT: "True"
```

### 2. Rich metrics dashboard (lowlighter/metrics)
Add `METRICS_TOKEN` secret (classic PAT with `read:user` + `repo` scopes), then create `.github/workflows/metrics.yml`:

```yaml
name: Metrics
on:
  schedule: [{ cron: "0 0 * * *" }]
  workflow_dispatch:
jobs:
  github-metrics:
    runs-on: ubuntu-latest
    permissions: { contents: write }
    steps:
      - uses: lowlighter/metrics@latest
        with:
          token: ${{ secrets.METRICS_TOKEN }}
          user: ArjunKalirana
          template: classic
          base: header, activity, repositories
          plugin_leetcode: yes
          plugin_leetcode_user: ArjunKal
          config_timezone: Asia/Kolkata
```

### 3. Stat card customization
All `github-readme-stats` links use `theme=tokyonight`. Swap for: `radical`, `dracula`, `nightowl`, `onedark`, `cobalt`, `synthwave`, `highcontrast`.

### 4. LeetCode card
Uses `leetcard.jacoblin.cool/ArjunKal` — just replace `ArjunKal` with your LeetCode username.
